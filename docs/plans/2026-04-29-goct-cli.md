```go
// Package config 加载并合并 goct 配置：flag > GOCT_* env > ~/.goct.yaml。
package config

import (
	"fmt"

	"github.com/spf13/viper"
)

type Resolved struct {
	URL, Username, Password, Cluster, Source string
	Insecure                                 bool
}

type Override struct {
	URL, Username, Password, Cluster, Source, ConfigFile string
	Insecure, InsecureSet                                bool
}

func Resolve(o Override) (Resolved, error) {
	v := viper.New()
	v.SetEnvPrefix("GOCT")
	v.AutomaticEnv()
	for _, k := range []string{"url", "username", "password", "insecure", "cluster", "source"} {
		_ = v.BindEnv(k)
	}
	v.SetDefault("source", "local")

	if o.ConfigFile != "" {
		v.SetConfigFile(o.ConfigFile)
		if err := v.ReadInConfig(); err != nil {
			return Resolved{}, fmt.Errorf("read %s: %w", o.ConfigFile, err)
		}
	} else {
		v.SetConfigName(".goct")
		v.SetConfigType("yaml")
		v.AddConfigPath("$HOME")
		_ = v.ReadInConfig()
	}

	r := Resolved{
		URL:      pick(o.URL, v.GetString("url")),
		Username: pick(o.Username, v.GetString("username")),
		Password: pick(o.Password, v.GetString("password")),
		Cluster:  pick(o.Cluster, v.GetString("cluster")),
		Source:   pick(o.Source, v.GetString("source")),
	}
	if o.InsecureSet {
		r.Insecure = o.Insecure
	} else {
		r.Insecure = v.GetBool("insecure")
	}
	return r, nil
}

func pick(xs ...string) string {
	for _, x := range xs {
		if x != "" {
			return x
		}
	}
	return ""
}
```

```go
// Package session 缓存 CloudTower 登录 token，避免每次命令都登录。
package session

import (
	"crypto/sha1"
	"encoding/hex"
	"encoding/json"
	"errors"
	"fmt"
	"os"
	"path/filepath"
	"time"
)

type Token struct {
	Value    string    `json:"value"`
	ExpireAt time.Time `json:"expire_at"`
}

func (t Token) Expired() bool { return time.Now().After(t.ExpireAt) }

func PathFor(host, user string) string {
	sum := sha1.Sum([]byte(host + "|" + user))
	return filepath.Join(cacheDir(), "session-"+hex.EncodeToString(sum[:])+".json")
}

func Save(host, user string, tok Token) error {
	p := PathFor(host, user)
	if err := os.MkdirAll(filepath.Dir(p), 0o700); err != nil {
		return fmt.Errorf("mkdir: %w", err)
	}
	data, _ := json.Marshal(tok)
	if err := os.WriteFile(p, data, 0o600); err != nil {
		return fmt.Errorf("write %s: %w", p, err)
	}
	return nil
}

func Load(host, user string) (Token, error) {
	data, err := os.ReadFile(PathFor(host, user))
	if err != nil {
		return Token{}, err
	}
	var tok Token
	if err := json.Unmarshal(data, &tok); err != nil {
		return Token{}, fmt.Errorf("decode: %w", err)
	}
	if tok.Expired() {
		return Token{}, errors.New("session token expired")
	}
	return tok, nil
}

func Delete(host, user string) error {
	err := os.Remove(PathFor(host, user))
	if err != nil && !os.IsNotExist(err) {
		return err
	}
	return nil
}

func cacheDir() string {
	if x := os.Getenv("XDG_CACHE_HOME"); x != "" {
		return filepath.Join(x, "goct")
	}
	if h, err := os.UserHomeDir(); err == nil {
		return filepath.Join(h, ".cache", "goct")
	}
	return filepath.Join(os.TempDir(), "goct")
}
```

```go
// Package output 提供统一的 table / JSON 渲染。
package output

import (
	"encoding/json"
	"fmt"
	"io"

	"github.com/olekukonko/tablewriter"
)

type Column struct {
	Header string
	Get    func(any) string
}

func Render(w io.Writer, data any, format string, columns []Column) error {
	switch format {
	case "json":
		enc := json.NewEncoder(w)
		enc.SetIndent("", "  ")
		return enc.Encode(data)
	case "table", "":
		return renderTable(w, data, columns)
	default:
		return fmt.Errorf("unsupported format: %s", format)
	}
}

func renderTable(w io.Writer, data any, columns []Column) error {
	rows, ok := data.([]any)
	if !ok {
		return fmt.Errorf("table render expects []any, got %T", data)
	}
	tw := tablewriter.NewWriter(w)
	headers := make([]string, len(columns))
	for i, c := range columns {
		headers[i] = c.Header
	}
	tw.SetHeader(headers)
	tw.SetAutoWrapText(false)
	tw.SetBorder(false)
	for _, item := range rows {
		row := make([]string, len(columns))
		for i, c := range columns {
			row[i] = c.Get(item)
		}
		tw.Append(row)
	}
	tw.Render()
	return nil
}
```

```go
import "fmt"

func HumanBytes(n uint64) string {
	const unit = 1024
	if n < unit {
		return fmt.Sprintf("%d B", n)
	}
	div, exp := uint64(unit), 0
	for x := n / unit; x >= unit; x /= unit {
		div *= unit
		exp++
	}
	return fmt.Sprintf("%.1f %s", float64(n)/float64(div),
		[]string{"KiB", "MiB", "GiB", "TiB", "PiB"}[exp])
}
```

```go
// Package task 提供异步任务进度的轮询与展示。
package task

import (
	"context"
	"errors"
	"fmt"
	"io"
	"os"
	"time"
)

var ErrFailed = errors.New("task failed")

// Ops 是 watcher 依赖的最小接口（由 pkg/adapter 实现）。
type Ops interface {
	GetTaskProgress(ctx context.Context, id string) (int, string, error)
}

type Options struct {
	Out      io.Writer
	Interval time.Duration
	Quiet    bool
}

type Watcher struct {
	ops  Ops
	opts Options
}

func New(ops Ops, opts Options) *Watcher {
	if opts.Out == nil {
		opts.Out = os.Stderr
	}
	if opts.Interval <= 0 {
		opts.Interval = time.Second
	}
	return &Watcher{ops: ops, opts: opts}
}

// Watch 阻塞轮询直到 task 终结或 ctx 取消。
func (w *Watcher) Watch(ctx context.Context, id string) error {
	t := time.NewTicker(w.opts.Interval)
	defer t.Stop()
	for {
		percent, status, err := w.ops.GetTaskProgress(ctx, id)
		if err != nil {
			return fmt.Errorf("query task %s: %w", id, err)
		}
		if !w.opts.Quiet {
			fmt.Fprintf(w.opts.Out, "\rTask %s: %d%% (%s)   ", id, percent, status)
		}
		switch status {
		case "SUCCESSED", "SUCCEEDED":
			if !w.opts.Quiet {
				fmt.Fprintln(w.opts.Out)
			}
			return nil
		case "FAILED":
			if !w.opts.Quiet {
				fmt.Fprintln(w.opts.Out)
			}
			return fmt.Errorf("task %s: %w", id, ErrFailed)
		}
		select {
		case <-ctx.Done():
			return ctx.Err()
		case <-t.C:
		}
	}
}
```

> 注：状态字符串以 T2 速查表为准；预留 SUCCESSED 与 SUCCEEDED 双拼写兼容。

```go
var (
	ErrAuth       = errors.New("authentication failed")
	ErrNotFound   = errors.New("not found")
	ErrTaskFailed = errors.New("task failed")

// Client 聚合接口，T7+ 通过内嵌追加各资源 sub-interface。
type Client interface {
	About(ctx context.Context) (TowerInfo, error)
}

type Options struct {
	URL, Username, Password, Source string
	Insecure                        bool
	Token                           string
}

type defaultClient struct {
	api       *apiclient.Cloudtower
	transport *httptransport.Runtime
}

// NewClient 构造 adapter.Client；提供 Token 时跳过登录。
func NewClient(_ context.Context, opts Options) (Client, SessionToken, error) {
	host, basePath, schemes, err := splitURL(opts.URL)
	if err != nil {
		return nil, SessionToken{}, err
	}
	tr := newTransport(host, basePath, schemes, opts.Insecure)

	if opts.Token != "" {
		tr.DefaultAuthentication = httptransport.APIKeyAuth("Authorization", "header", opts.Token)
		return &defaultClient{api: apiclient.New(tr, strfmt.Default), transport: tr},
			SessionToken{Value: opts.Token}, nil
	}

	api := apiclient.New(tr, strfmt.Default)
	tok, err := login(api, tr, opts)
	if err != nil {
		return nil, SessionToken{}, err
	}
	return &defaultClient{api: api, transport: tr}, tok, nil
}

func login(api *apiclient.Cloudtower, tr *httptransport.Runtime, opts Options) (SessionToken, error) {
	src := models.UserSourceLOCAL
	if opts.Source != "" && opts.Source != "local" {
		src = models.UserSource(opts.Source)
	}
	p := userPkg.NewLoginParams()
	p.RequestBody = &models.LoginInput{
		Username: pointy.String(opts.Username),
		Password: pointy.String(opts.Password),
		Source:   models.NewUserSource(src),
	}
	res, err := api.User.Login(p)
	if err != nil {
		return SessionToken{}, fmt.Errorf("%w: %v", ErrAuth, err)
	}
	if res.Payload == nil || res.Payload.Data == nil || res.Payload.Data.Token == nil {
		return SessionToken{}, fmt.Errorf("%w: empty token", ErrAuth)
	}
	tr.DefaultAuthentication = httptransport.APIKeyAuth("Authorization", "header", *res.Payload.Data.Token)
	return SessionToken{Value: *res.Payload.Data.Token, ExpireAt: time.Now().Add(8 * time.Hour)}, nil
}

func newTransport(host, basePath string, schemes []string, insecure bool) *httptransport.Runtime {
	if insecure {
		hc := &http.Client{Transport: &http.Transport{TLSClientConfig: &tls.Config{InsecureSkipVerify: true}}}
		return httptransport.NewWithClient(host, basePath, schemes, hc)
	}
	return httptransport.New(host, basePath, schemes)
}

func splitURL(raw string) (host, basePath string, schemes []string, err error) {
	u, err := url.Parse(raw)
	if err != nil || u.Host == "" {
		return "", "", nil, fmt.Errorf("invalid URL %q: %w", raw, err)
	}
	return u.Host, "/v2/api", []string{u.Scheme}, nil
}
```

```go
import (
	"context"
```

```go
func New(ctx context.Context, cfg config.Resolved) (adapter.Client, error) {
	if cfg.URL == "" {
		return nil, errors.New("missing --url / GOCT_URL")
	}
	host := hostKey(cfg.URL)

	if tok, err := session.Load(host, cfg.Username); err == nil {
		c, _, e := adapter.NewClient(ctx, adapter.Options{URL: cfg.URL, Insecure: cfg.Insecure, Token: tok.Value})
		if e == nil {
			return c, nil
		}
		_ = session.Delete(host, cfg.Username)
	}

	c, tok, err := adapter.NewClient(ctx, adapter.Options{
		URL: cfg.URL, Username: cfg.Username, Password: cfg.Password,
		Source: cfg.Source, Insecure: cfg.Insecure,
	})
	if err != nil {
		return nil, err
	}
	_ = session.Save(host, cfg.Username, session.Token{Value: tok.Value, ExpireAt: tok.ExpireAt})
	return c, nil
}

func hostKey(raw string) string {
	if u, err := url.Parse(raw); err == nil && u.Host != "" {
		return u.Host
	}
	return raw
}

type ctxKey struct{}

func With(ctx context.Context, c adapter.Client) context.Context {
	return context.WithValue(ctx, ctxKey{}, c)
}

func From(ctx context.Context) adapter.Client {
	c, _ := ctx.Value(ctxKey{}).(adapter.Client)
	return c
}
```

```go
package service

import (
	"context"
	"fmt"
	"regexp"

	"github.com/6547709/goct/pkg/adapter"
)

var uuidRe = regexp.MustCompile(`^[0-9a-fA-F-]{36}$`)

func Resolve[T any](
	ctx context.Context,
	list func(ctx context.Context, opts adapter.ListOpts) ([]T, error),
	get func(ctx context.Context, id string) (*T, error),
	extract func(T) (id, name string),
	idOrName string,
) (*T, error) {
	if uuidRe.MatchString(idOrName) {
		return get(ctx, idOrName)
	}
	all, err := list(ctx, adapter.ListOpts{NameContains: idOrName})
	if err != nil {
		return nil, err
	}
	matches := make([]T, 0, 1)
	for _, v := range all {
		_, n := extract(v)
		if n == idOrName {
			matches = append(matches, v)
		}
	}
	switch len(matches) {
	case 0:
		return nil, fmt.Errorf("%w: %q", adapter.ErrNotFound, idOrName)
	case 1:
		return &matches[0], nil
	default:
		return nil, fmt.Errorf("ambiguous name %q (%d matches)", idOrName, len(matches))
	}
}
```

```go
			return output.Render(c.OutOrStdout(), items, "table", output.VMListColumns)
		},
	}
	sf.Register(c)
	return c
}
```

```go
package vm

import (
	"errors"

	"github.com/6547709/goct/pkg/adapter"
	"github.com/6547709/goct/pkg/client"
	"github.com/6547709/goct/pkg/service"
	"github.com/6547709/goct/pkg/task"
	"github.com/spf13/cobra"
)

func newPowerOn() *cobra.Command {
	return &cobra.Command{
		Use: "vm.power.on <name|id>", Short: "Power on a VM", GroupID: groupID,
		Args: cobra.ExactArgs(1),
		RunE: func(c *cobra.Command, args []string) error {
			cli := client.From(c.Context())
			svc := service.NewVM(cli)
			vm, err := svc.Resolve(c.Context(), args[0])
			if err != nil {
				return err
			}
			ref, err := svc.Power(c.Context(), vm.ID, adapter.PowerOn, false)
			if err != nil {
				return err
			}
			if ref.ID == "" {
				return errors.New("server returned no task ID")
			}
			w := task.New(cli, task.Options{}) // adapter.Client 实现 task.Ops
			return w.Watch(c.Context(), ref.ID)
		},
	}
}
```
